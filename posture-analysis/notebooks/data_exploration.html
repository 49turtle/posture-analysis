# AI-Hub JSON 파일 로드 함수
def load_ai_hub_json(path):
    with open(path, 'r', encoding='utf-8') as f:
        data = json.load(f)
    keypoints = []
    for annotation in data['annotations']:
        keypoints.append(annotation['keypoints'])
    return keypoints

# OpenPose JSON 파일 로드 함수
def load_openpose_json(path):
    with open(path, 'r', encoding='utf-8') as f:
        data = json.load(f)
    keypoints = []
    for frame in data:
        for person in frame['keypoints']:
            keypoints.append(person)
    return keypoints

# 시각화 함수 정의
def visualize_ai_hub_keypoints(keypoints, labels, title="Keypoints"):
    for person in keypoints[:1]:  # 첫 번째 사람만 시각화
        x = person[0::3]
        y = person[1::3]
        # (0, 0) 좌표 필터링
        x_filtered = [x_i for x_i, y_i in zip(x, y) if not (x_i == 0 and y_i == 0)]
        y_filtered = [y_i for x_i, y_i in zip(x, y) if not (x_i == 0 and y_i == 0)]
        labels_filtered = [label for x_i, y_i, label in zip(x, y, labels) if not (x_i == 0 and y_i == 0)]
        
        plt.scatter(x_filtered, y_filtered)
        for i, label in enumerate(labels_filtered):
            plt.text(x_filtered[i], y_filtered[i], label, fontsize=8, ha='right')
    plt.gca().invert_yaxis()  # 이미지 좌표계를 고려하여 y축을 반전
    plt.title(title)
    plt.xlabel("x")
    plt.ylabel("y")
    plt.show()

# AI-Hub 시각화 함수 정의
def visualize_ai_hub_keypoints_with_lines(keypoints, labels, connections, title="Keypoints with Connections"):
    for person in keypoints[:1]:  # 첫 번째 사람만 시각화
        x = person[0::3]
        y = person[1::3]
        # (0, 0) 좌표 필터링
        x_filtered = [x_i for x_i, y_i in zip(x, y) if not (x_i == 0 and y_i == 0)]
        y_filtered = [y_i for x_i, y_i in zip(x, y) if not (x_i == 0 and y_i == 0)]
        labels_filtered = [label for x_i, y_i, label in zip(x, y, labels) if not (x_i == 0 and y_i == 0)]
        index_filtered = [i for i, (x_i, y_i) in enumerate(zip(x, y)) if not (x_i == 0 and y_i == 0)]
        
        plt.scatter(x_filtered, y_filtered)
        for i, label in enumerate(labels_filtered):
            plt.text(x_filtered[i], y_filtered[i], label, fontsize=8, ha='right')
        
        for (i, j) in connections:
            if i in index_filtered and j in index_filtered:
                xi, yi = x[i], y[i]
                xj, yj = x[j], y[j]
                plt.plot([xi, xj], [yi, yj], 'r-')
        
    plt.gca().invert_yaxis()  # 이미지 좌표계를 고려하여 y축을 반전
    plt.title(title)
    plt.xlabel("x")
    plt.ylabel("y")
    plt.show()
